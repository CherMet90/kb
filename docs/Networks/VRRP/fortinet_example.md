```
config system interface
    edit "port2"
        config vrrp
            edit 1                          # можно создавать несколько сущностей на одном интерфейсе с разными IP, например, для реализации балансировки
                set vrip 198.51.100.3       # виртуальный ip
                set priority 120            # задаём приоритет от 1 до 255 для выбора мастер-устройства. Значение по-умолчанию - 100. Мастером становится устройство с большим приоритетом
                set vrdst 1.1.1.1 8.8.8.8   # реализация failover. Если адреса из списка недоступны, роутер перестаёт анонсировать себя как активный роутер
                set vrdst-priority 90       # этот приоритет присвоит себе роутер при недоступности failover-адресов
            next
            edit 2
                set vrip 198.51.100.4
                set priority 110
                set vrdst 1.1.1.1 8.8.8.8
                set vrdst-priority 90
            next
        end
```
